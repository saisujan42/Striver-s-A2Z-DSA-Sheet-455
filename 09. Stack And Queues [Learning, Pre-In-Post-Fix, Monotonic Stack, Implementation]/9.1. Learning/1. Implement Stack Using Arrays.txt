1. Implement Stack Using Arrays
TC : O(1) SC : O(1)

/*

{
private:
    int arr[1000];
    int top;

    MyStack(){top=-1;}

void MyStack ::push(int x) 
{
    arr[++top] = x;
}
int MyStack ::pop() 
{
    if(top == -1)
        return -1;
    return arr[top--];
}
