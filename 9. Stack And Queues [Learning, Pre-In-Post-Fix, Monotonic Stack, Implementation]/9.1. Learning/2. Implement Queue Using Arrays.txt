1. Implement Queue Using Array
TC : O(1) SC : O(1)

/* 

The structure of the class is
class MyQueue {
private:
    int arr[100005];
    int front;
    int rear;

public :
    MyQueue(){front=0;rear=0;}
    void push(int);
    int pop();
};
 */

//Function to push an element x in a queue.
void MyQueue :: push(int x)
{
    arr[rear++] = x; 
}
int MyQueue :: pop()
{
    if(rear <= front) return -1;
    return arr[front++];
}
